输入一个正整数，一个负整数，组合成长度为17的数组，使得连续的7个数的和为0， 求最大的和？

#include<iostream>
#define max(a,b) ((a)>(b)?(a):(b))
using namespace std;

inline bool isValid(unsigned flag, int pos, int neg){
    int s = 0;
    int mask = 1;
    for(int i=0; i<7 ; i++){
        s += flag & mask ? pos : neg;
        mask  = mask << 1;
    }
    if(s >= 0)
        return false;

    for(int i=7; i<17; i++) {
        s += flag & mask ? pos : neg;
        s -= flag & (mask >> 7) ? pos : neg;
        mask = mask << 1;
        if (s >= 0)
            return false;
    }
    return true;
}
inline int getSum(unsigned flag, int pos, int neg) {
    int s = 0;
    int mask = 1;
    for(int i=0; i<17; i++) {
        s += flag & mask ? pos : neg;
        mask = mask << 1;
    }
    return s;
}
int main(){
    int pos=10, neg=-61;
    unsigned end = 1<<17;
    int max_s = 0x80000000;
    isValid(131039, pos, neg);
    for(unsigned flag = 0; flag < end; flag++){
        if(isValid(flag, pos, neg)){
            int s = getSum(flag, pos, neg);
            if(s > max_s){
                max_s = s;
                printf("flag = %u s = %d\n",flag, max_s);
            }

        }
    }

}
